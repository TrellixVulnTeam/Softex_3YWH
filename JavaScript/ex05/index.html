var readlineSync = require('readline-sync'); var teste = readlineSync.question('\n\n Digite a operacao que deseja realizar.\n Considere - \n + \n - \n * \n / \n ** \n\n'); var n1 = parseFloat(readlineSync.question('Digite um numero: ')); var n2 = parseFloat(readlineSync.question('Digite outro numero: ')); if(teste == '+'){ console.log(n1+n2); }else if(teste == '-'){ console.log(n1-n2); } else if(teste == '*'){ console.log(n1*n2); } else if(teste == '/'){ console.log('Divisao inteira: ' + Math.floor(n1/n2)); if(n1%n2 != 0){ console.log('e o resto da divisao: '+(n1%n2)); } } else if(teste == '**'){ console.log(n1**n2); }else{ console.log('Passe Valores Validos para utilizar a Calculadora.'); }
